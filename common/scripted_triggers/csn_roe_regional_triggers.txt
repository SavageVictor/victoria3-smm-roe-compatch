# Subject Management Menu - RoE Compatch: Regional Triggers
# Maps vanilla SMM regional categories to RoE's continent-level geographic regions
# Based on RoE's common/geographic_regions/roe_geographic_regions.txt

csn_state_is_in_africa = {
    OR = {
        region = sr:region_africa
    }
}

csn_state_is_in_asia = {
    OR = {
        region = sr:region_china
        region = sr:region_east_asia
        region = sr:region_south_east_asia
        region = sr:region_dornod
    }
}
