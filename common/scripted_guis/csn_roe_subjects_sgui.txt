# Subject Management Menu - RoE Compatch: Subject Type GUI Entries

csn_roe_subject_banner = {
    scope = country
    is_shown = {
        is_player = yes
        NOT = { has_variable = csn_roe_sel_banner }
    }
    effect = {
        remove_variable_list = { name = csn_roe_sel_banner name = csn_roe_sel_march name = csn_roe_sel_letter_of_marque name = csn_roe_sel_real_union name = csn_roe_sel_satrapy name = csn_roe_sel_subsidiary }
        set_variable = csn_roe_sel_banner
    }
}

    is_shown = {

}
