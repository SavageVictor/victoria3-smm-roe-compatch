csn_select_this_mil_affairs_sgui = {
    scope = country
    is_shown = {
        has_variable = csn_select_this_mil_affairs_var
    }
    effect = {
        # 选中标记
        overlord = {
            if = {
                limit = { has_variable_list = csn_mil_affairs_list }
                every_in_list = {
                    variable = csn_mil_affairs_list
                    limit = { has_variable = csn_select_this_mil_affairs_var }
                    remove_variable = csn_select_this_mil_affairs_var
                }
            }
        }
        set_variable = csn_select_this_mil_affairs_var


        # 选中国家的scope。。。。。。
        overlord = {
            if = {
                limit = { has_variable_list = csn_mil_affairs_list }
                every_in_list = {
                    variable = csn_mil_affairs_list
                    limit = { has_variable = csn_select_this_mil_affairs_var }
                    save_scope_as = csn_select_this_mil_affairs_scope
                }
            }
            set_variable = {
                name = csn_select_this_mil_affairs_var_fixed
                value = scope:csn_select_this_mil_affairs_scope
            }
        }



    }
}
